## Importing Components

```tsx
import { For } from "@zayne-labs/ui-react/common/for";
import { DragScroll } from "@zayne-labs/ui-react/ui/drag-scroll";
```

Utility components live under `/common/`, UI components under `/ui/`.

## Examples

### Conditional Rendering

```tsx
import { Switch } from "@zayne-labs/ui-react/common/switch";

function StatusView({ status }: { status: string }) {
	return (
		<Switch.Root value={status}>
			<Switch.Match when="loading">Loading...</Switch.Match>
			<Switch.Match when="error">Something went wrong</Switch.Match>
			<Switch.Default>Content loaded!</Switch.Default>
		</Switch.Root>
	);
}
```

### List Rendering

```tsx
import { For } from "@zayne-labs/ui-react/common/for";

function UserList({ users }: { users: { id: number; name: string }[] }) {
	return (
		<For each={users} fallback={<p>No users found</p>}>
			{(user) => <div key={user.id}>{user.name}</div>}
		</For>
	);
}
```

### Drag to Scroll

```tsx
import { DragScroll } from "@zayne-labs/ui-react/ui/drag-scroll";

function HorizontalList() {
	return (
		<DragScroll.Root>
			<DragScroll.List className="flex gap-4 p-4">
				{[...Array(5).keys()].map((number) => (
					<DragScroll.Item
						key={number}
						className="flex h-40 w-64 shrink-0 items-center justify-center rounded-xl border"
					>
						Item {number + 1}
					</DragScroll.Item>
				))}
			</DragScroll.List>

			<DragScroll.Prev>←</DragScroll.Prev>
			<DragScroll.Next>→</DragScroll.Next>
		</DragScroll.Root>
	);
}
```
