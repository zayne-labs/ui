## Basic Usage

```tsx
import { Carousel } from "@zayne-labs/ui-react/ui/carousel";

const items = [
	{ src: "/image1.jpg", title: "Slide 1" },
	{ src: "/image2.jpg", title: "Slide 2" },
	{ src: "/image3.jpg", title: "Slide 3" },
];

<Carousel.Root images={items} hasAutoSlide={true} autoSlideInterval={5000} shouldPauseOnHover={true}>
	<Carousel.ItemList>
		{({ image }) => (
			<Carousel.Item>
				<img src={image.src} alt={image.title} />
			</Carousel.Item>
		)}
	</Carousel.ItemList>

	<Carousel.Controls />

	<Carousel.IndicatorList>
		{({ index }) => <Carousel.Indicator currentIndex={index} />}
	</Carousel.IndicatorList>
</Carousel.Root>;
```

## Navigation Icons

<Accordions>
	<Accordion title="Separate Icons">
		```tsx
		<Carousel.Controls
			icon={{
				prev: <PrevIcon />,
				next: <NextIcon />,
			}}
		/>
		```
	</Accordion>
	<Accordion title="Single Icon (Auto-Rotate)">
		```tsx
		<Carousel.Controls
			icon={{

				iconType: "nextIcon", // Prev button rotates 180deg
			}}
		/>
		```
	</Accordion>
