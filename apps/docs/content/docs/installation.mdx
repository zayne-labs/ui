import { Step, Steps } from "fumadocs-ui/components/steps";

<Steps>
<Step>

### Install the Package

```package-install
@zayne-labs/ui-react
```

</Step>
<Step>

### Peer Dependencies

| Dependency   | Version                |
| :----------- | :--------------------- |
| `react`      | `^19.0.0`              |
| `react-dom`  | `^19.0.0`              |
| `typescript` | `^5.0.0` (recommended) |

<Accordions>
  <Accordion title="Form component — react-hook-form">
    ```package-install
    react-hook-form
    ```
  </Accordion>

  <Accordion title="Tailwind CSS v4">
    ```package-install
    -D tailwindcss@latest
    ```
  </Accordion>
</Accordions>

</Step>
<Step>

### TypeScript

```json title="tsconfig.json"
{
	"compilerOptions": {
		"moduleResolution": "bundler",
		"jsx": "react-jsx",
		"strict": true
	}
}
```

</Step>
<Step>

### CSS Setup

<Tabs items={["Tailwind CSS v4", "Standalone CSS"]}>
	<Tab value="Tailwind CSS v4">
		```css title="app/globals.css"
		@import "tailwindcss";
		@import "@zayne-labs/ui-react/css/preset.css";
		```
	</Tab>

    <Tab value="Standalone CSS">
     ```tsx title="app/globals.css"
     import "@zayne-labs/ui-react/style.css";
     ```
     </Tab>

</Tabs>

<Callout type="info">
	The Tailwind preset is recommended — enables theme tokens, animations, and utilities. See the [Styling
	Guide](/docs/styling).
</Callout>

</Step>
</Steps>
